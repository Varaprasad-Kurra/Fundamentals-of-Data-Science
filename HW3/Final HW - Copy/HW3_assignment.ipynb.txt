    "# entropy (H)\n",
    "def entropy(feature, dataset):\n",
    "    values = dataset[feature].value_counts()\n",
    "    Total=0\n",
    "    for i in values:\n",
    "        Total = Total + i;\n",
    "\n",
    "    Total_Prob = 0\n",
    "    for i in values:\n",
    "        Indiviudal_Prob=(i/Total)*math.log(i/Total,2);             #Using the Entropy Formaule to calculate the Entropy\n",
    "        Total_Prob = Total_Prob + Indiviudal_Prob\n",
    "        Entropy = -round((Total_Prob),2)\n",
    "    print('\\n')\n",
    "    return Entropy\n",

    "# information gain (IG)\n",
    "import numpy as np\n",
    "def InfoGain(feature, target, dataset, measure):\n",
    "    \n",
    "    if measure == \"entropy\":\n",
    "        Target_Entropy = entropy(target,dataset)\n",
    "        values,counts= np.unique(dataset[feature],return_counts=True)\n",
    "        p_Entropy=0\n",
    "        for i in range(len(values)):\n",
    "                       # To find weighted gini we make usethe Gain_Dataset Gain_Dataset = dataset[dataset[feature] == values[i]]\n",
    "            weighted_entropy = np.sum(counts[i]/np.sum(counts)*entropy(target,dataset[dataset[feature] == values[i]]))\n",
    "            p_Entropy = p_Entropy + weighted_entropy\n",
    "        print('\\n')\n",
    "        IG =  round(Target_Entropy - p_Entropy,6)\n",
    "        print('IG from Entropy withrespect to attribute ',feature,' is ', IG)\n",
    "        return IG\n",
    "   \n",
    "    else:\n",
    "        \n",
    "        Target_Gini = gini(target, dataset)\n",
    "        values,counts= np.unique(dataset[feature],return_counts=True)\n",
    "        probability_gini=0\n",
    "        for i in range(len(values)):\n",
    "           # To find weighted gini we make usethe Gain_Dataset Gain_Dataset = dataset[dataset[feature] == values[i]]\n",
    "            weighted_Gini = (np.sum(counts[i]/np.sum(counts)*gini(target,dataset[dataset[feature] == values[i]])))\n",
    "            probability_gini = probability_gini + weighted_Gini\n",
    "        print('\\n')\n",
    "        IG =  round(Target_Gini - probability_gini,6)\n",
    "        print('IG from gini withrespect to the attribute ',feature,' is ', IG)\n",
    "        return IG\n",
